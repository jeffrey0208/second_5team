-- 멤버 테이블 및 시퀀스

CREATE TABLE MEMBER(
MEM_NUM NUMBER(20) PRIMARY KEY,
MEM_ID VARCHAR2(20) NOT NULL,
MEM_PW VARCHAR2(20) NOT NULL,
MEM_NAME VARCHAR2(20) NOT NULL,
MEM_BIRTH DATE NOT NULL,
MEM_GEN VARCHAR2(8) NOT NULL,
MEM_EMAIL VARCHAR2(50) NOT NULL,
MEM_PHONE VARCHAR2(11) NOT NULL,
MEM_ADD1 VARCHAR2(200) NOT NULL,
MEM_JOINDATE DATE DEFAULT SYSDATE NOT NULL,
MEM_VERIFY VARCHAR2(2) DEFAULT 'Y' NOT NULL,
MEM_LEVEL NUMBER(2) DEFAULT 1 NOT NULL,
MEM_DEL_GB VARCHAR2(2) DEFAULT 'N' NOT NULL
);

CREATE SEQUENCE MEMBER_SEQ START WITH 1 INCREMENT BY 1 NOMAXVALUE MINVALUE 1;

-- 상품 게시판 테이블 및 시퀀스
